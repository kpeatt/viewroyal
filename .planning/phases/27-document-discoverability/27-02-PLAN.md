1. Add this function after the existing `getMatterById` function:

```typescript
export async function getDocumentsForAgendaItems(
  supabase: SupabaseClient,
  agendaItemIds: number[],
) {
  if (agendaItemIds.length === 0) return [];
  const { data, error } = await supabase
    .from("extracted_documents")
    .select("id, document_id, agenda_item_id, title, document_type, page_start, page_end")
    .in("agenda_item_id", agendaItemIds)
    .order("agenda_item_id")
    .order("page_start", { ascending: true, nullsFirst: false });
  if (error) {
    console.error("Error fetching documents for agenda items:", error);
    return [];
  }
  return data ?? [];
}
```

Key design decisions:
- Only fetches metadata columns (id, document_id, agenda_item_id, title, document_type, page_start, page_end) -- NO summary, key_facts, or section_text since the matter page only shows chips per CONTEXT.md decisions
- Returns empty array on error (graceful degradation, not crash)
- Orders by agenda_item_id then page_start for consistent grouping
- Does NOT modify the existing `getMatterById` function (that select string is already extremely long)

2. Update the loader to fetch documents after getting the matter. Since we need agenda item IDs from the matter response first, this is a sequential call (unavoidable data dependency, but the `.in()` query is fast):

```typescript
export async function loader({ params }: Route.LoaderArgs) {
  const { id } = params;
  try {
    const supabase = getSupabaseAdminClient();
    const matter = await getMatterById(supabase, id);

    // Fetch documents for all agenda items in this matter (batch query)
    const agendaItemIds = (matter.agenda_items || []).map((ai) => ai.id);
    const matterDocuments = await getDocumentsForAgendaItems(supabase, agendaItemIds);

    return { matter, matterDocuments };
  } catch (error) {
    console.error("Error fetching matter:", error);
    throw new Response("Matter Not Found", { status: 404 });
  }
}
```
